export class AutoInit {
  static components = {
    'dropdown': async (el, options) => {
      const Dropdown = await IMCAT.use('dropdown');
      return new Dropdown(el, options);
    },
    'modal': async (el, options) => {
      const Overlays = await IMCAT.use('overlays');
      const modal = new Overlays.Modal(options);
      el.addEventListener('click', () => modal.show());
      return modal;
    },
    'tooltip': async (el, options) => {
      const Tooltips = await IMCAT.use('tooltips');
      return new Tooltips.Tooltip(el, options);
    }
  };
  
  static init() {
    document.querySelectorAll('[data-imcat]').forEach(async (el) => {
      const component = el.dataset.imcat;
      const options = this._parseOptions(el);
      
      if (this.components[component]) {
        const instance = await this.components[component](el, options);
        IMCAT.view.registerInstance(instance);
      }
    });
  }
  
  static _parseOptions(el) {
    const options = {};
    Object.keys(el.dataset).forEach(key => {
      if (key !== 'imcat') {
        try {
          options[key] = JSON.parse(el.dataset[key]);
        } catch {
          options[key] = el.dataset[key];
        }
      }
    });
    return options;
  }

  formData(selector) {
    const form = typeof selector === 'string' 
      ? document.querySelector(selector) 
      : selector;
    const formData = new FormData(form);
    const data = {};
    formData.forEach((value, key) => {
      if (data[key]) {
        if (!Array.isArray(data[key])) data[key] = [data[key]];
        data[key].push(value);
      } else {
        data[key] = value;
      }
    });
    return data;

  fillForm(selector, data) {
    const form = typeof selector === 'string' 
      ? document.querySelector(selector) 
      : selector;
    Object.entries(data).forEach(([key, value]) => {
      const input = form.elements[key];
      if (input) {
        if (input.type === 'checkbox') {
          input.checked = Boolean(value);
        } else if (input.type === 'radio') {
          const radio = form.querySelector(`[name="${key}"][value="${value}"]`);
          if (radio) radio.checked = true;
        } else {
          input.value = value;
        }
      }
    });

      // í´ë°±
      const textarea = document.createElement('textarea');
      textarea.value = text;
      document.body.appendChild(textarea);
      textarea.select();
      document.execCommand('copy');
      textarea.remove();

  },

// src/core/formatters.js
export const Formatters = {
  number(value, locale = 'ko-KR') {
    return new Intl.NumberFormat(locale).format(value);
  },
  
  currency(value, currency = 'KRW', locale = 'ko-KR') {
    return new Intl.NumberFormat(locale, { 
      style: 'currency', 
      currency 
    }).format(value);
  },
  
  date(value, options = {}) {
    const date = value instanceof Date ? value : new Date(value);
    const defaults = { year: 'numeric', month: '2-digit', day: '2-digit' };
    return new Intl.DateTimeFormat('ko-KR', { ...defaults, ...options }).format(date);
  },
  
  time(value, options = {}) {
    const date = value instanceof Date ? value : new Date(value);
    const defaults = { hour: '2-digit', minute: '2-digit' };
    return new Intl.DateTimeFormat('ko-KR', { ...defaults, ...options }).format(date);
  },
  
  datetime(value) {
    return `${this.date(value)} ${this.time(value)}`;
  },
  
  relative(value) {
    const date = value instanceof Date ? value : new Date(value);
    const rtf = new Intl.RelativeTimeFormat('ko-KR', { numeric: 'auto' });
    const diff = (date - new Date()) / 1000;
    
    if (Math.abs(diff) < 60) return rtf.format(Math.round(diff), 'second');
    if (Math.abs(diff) < 3600) return rtf.format(Math.round(diff / 60), 'minute');
    if (Math.abs(diff) < 86400) return rtf.format(Math.round(diff / 3600), 'hour');
    if (Math.abs(diff) < 2592000) return rtf.format(Math.round(diff / 86400), 'day');
    return this.date(date);
  },
  
  bytes(bytes, decimals = 2) {
    if (bytes === 0) return '0 Bytes';
    const k = 1024;
    const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(decimals)) + ' ' + sizes[i];
  },
  
  truncate(str, maxLength, suffix = '...') {
    if (str.length <= maxLength) return str;
    return str.substring(0, maxLength - suffix.length) + suffix;
  },
  
  phone(value) {
    const cleaned = value.replace(/\D/g, '');
    if (cleaned.length === 11) {
      return cleaned.replace(/(\d{3})(\d{4})(\d{4})/, '$1-$2-$3');
    }
    return value;
  }
};

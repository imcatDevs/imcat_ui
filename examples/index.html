빠른 시작

// 1. 파일 로드
<link rel="stylesheet" href="dist/imcat-ui.css">
<script type="module" src="dist/imcat-ui.min.js"></script>

// 2. 단축 API 사용
await IMCAT.alert('Hello World!');

if (await IMCAT.confirm('계속하시겠습니까?')) {
  IMCAT.toast.success('완료!');
}

// 3. 모듈 동적 로드
const Overlays = await IMCAT.use('overlays');
const modal = new Overlays.Modal({ 
  title: '제목',
  content: '내용'
});
modal.show();

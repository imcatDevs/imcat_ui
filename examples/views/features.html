touch_app DOM 조작

jQuery 스타일의 친숙한 API로 DOM을 쉽게 조작할 수 있습니다. 

// 요소 선택
const el = IMCAT.dom('#myElement');

// 내용 변경
el.html('<strong>New Content</strong>');
el.text('Plain Text');

// 클래스 조작
el.addClass('active');
el.removeClass('hidden');
el.toggleClass('show');

// 스타일 설정
el.css({ color: 'red', fontSize: '16px' });

route SPA 라우팅

History API 기반의 부드러운 페이지 전환을 지원합니다. 

// 라우트 정의
IMCAT.router.route('/home', 'views/home.html');
IMCAT.router.route('/about', 'views/about.html');

// 라우트 이동
IMCAT.router.navigate('/about');

// 라우트 훅
IMCAT.router.beforeLoad((path) => {
  console.log('Loading:', path);
});

IMCAT.router.afterLoad((path) => {
  console.log('Loaded:', path);
});

storage 상태 관리

리액티브 상태 관리와 computed 속성을 지원합니다. 

// 상태 생성
const state = IMCAT.createStore({
  firstName: 'John',
  lastName: 'Doe',
  age: 25
});

// Computed 속성
state.compute('fullName', ['firstName', 'lastName'], 
  (first, last) => `${first} ${last}`
);

// 상태 감시
state.watch('age', (newAge) => {
  console.log('나이가 변경되었습니다:', newAge);
});

// 배치 업데이트
state.batch(() => {
  state.firstName = 'Jane';
  state.age = 30;
}); // 한 번만 알림

event 이벤트 버스

컴포넌트 간 통신을 위한 전역 이벤트 시스템입니다. 

// 이벤트 구독
IMCAT.events.on('user:login', (data) => {
  console.log('사용자 로그인:', data);
});

// 이벤트 발생
IMCAT.events.emit('user:login', {
  username: 'john',
  timestamp: Date.now()
});

// 한 번만 실행
IMCAT.events.once('app:ready', () => {
  console.log('앱이 준비되었습니다!');
});

// 구독 해제
IMCAT.events.off('user:login', handler);

security XSS 방어

자동 이스케이프와 새니타이징으로 XSS 공격을 방어합니다. 

// 자동 이스케이프
const userInput = '<script>alert("XSS")</script>';
const safe = IMCAT.escape(userInput);
// → &lt;script&gt;alert("XSS")&lt;/script&gt;

// HTML 새니타이징
const html = '<div>Safe</div><script>Bad</script>';
const clean = IMCAT.sanitize(html);
// → <div>Safe</div>

// 경로 검증
IMCAT.security.validatePath('views/page.html'); // true
IMCAT.security.validatePath('../etc/passwd');   // false

animation 애니메이션

20가지 이상의 내장 애니메이션 효과를 제공합니다. 

// Fade In
IMCAT.animation.fadeIn('#element', 300);

// Slide Down
IMCAT.animation.slideDown('#panel', 400);

// Custom Animation
IMCAT.animation.custom('#box', {
  opacity: [0, 1],
  transform: ['translateY(-20px)', 'translateY(0)']
}, 500, 'ease-out');

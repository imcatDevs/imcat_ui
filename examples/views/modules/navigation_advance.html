TreeView 사용법

const Navigation = await IMCAT.use('navigation');

// TreeView 생성
const treeview = new Navigation.TreeView('#container', {
  multipleSelect: false,  // 다중 선택 허용
  expandIcon: 'chevron_right',
  onChange: (node, isExpanded) => {
    console.log('TreeView 변경:', node, isExpanded);
  }
});

// 노드 확장/축소
treeview.expandNode(nodeElement);
treeview.collapseNode(nodeElement);

// 모두 확장/축소
treeview.expandAll();
treeview.collapseAll();

// 노드 선택
treeview.selectNode(nodeElement);

// 정리
treeview.destroy();

다단계 수직 메뉴 사용법

// HTML 구조
<ul class="menu menu--vertical menu--nested">
  <li class="menu__item">
    <a href="#" class="menu__link">
      <i class="material-icons-outlined menu__icon">dashboard</i>
      <span class="menu__text">대시보드</span>
    </a>
  </li>
  
  <li class="menu__item has-submenu">
    <a href="#" class="menu__link">
      <i class="material-icons-outlined menu__icon">settings</i>
      <span class="menu__text">설정</span>
    </a>
    <ul class="menu__submenu">
      <li class="menu__item">
        <a href="#" class="menu__link">
          <span class="menu__text">일반 설정</span>
        </a>
      </li>
    </ul>
  </li>
</ul>

// JavaScript 초기화
function initNestedMenu(menuId) {
  const menu = document.getElementById(menuId);
  menu.querySelectorAll('.menu__item.has-submenu > .menu__link').forEach(link => {
    link.addEventListener('click', function(e) {
      e.preventDefault();
      this.parentElement.classList.toggle('is-open');
    });
  });
}

initNestedMenu('sidebar_menu');

MegaMenu 사용법

const Navigation = await IMCAT.use('navigation');

// MegaMenu 생성
const megamenu = new Navigation.MegaMenu('#megamenu', {
  trigger: 'hover',        // 'hover' 또는 'click'
  hoverDelay: 200,         // hover 지연 시간 (ms)
  closeOnOutside: true,    // 외부 클릭 시 닫기
  onChange: (index, isOpen, item) => {
    console.log('MegaMenu 변경:', index, isOpen);
  }
});

// 메뉴 열기/닫기
megamenu.openItem(0);
megamenu.closeItem(0);
megamenu.closeAll();

// 정리
megamenu.destroy();

통합 사용 예제

// Tabs + TreeView + Accordion + MegaMenu 통합
const tabs = new Navigation.Tabs('#sidebar-tabs', {
  orientation: 'vertical'
});

const treeview = new Navigation.TreeView('#file-tree', {
  multipleSelect: true
});

const accordion = new Navigation.Accordion('#settings', {
  multiple: false
});

const megamenu = new Navigation.MegaMenu('#main-menu', {
  trigger: 'hover',
  hoverDelay: 200
});

// 메모리 관리
IMCAT.view.registerInstance(tabs);
IMCAT.view.registerInstance(treeview);
IMCAT.view.registerInstance(accordion);
IMCAT.view.registerInstance(megamenu);

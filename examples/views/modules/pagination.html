API 사용법

Pagination 클래스
const Pagination = await IMCAT.use('pagination');

// 기본 사용
const pagination = new Pagination.Pagination('#container', {
  totalItems: 100,
  itemsPerPage: 10,
  currentPage: 1,
  maxVisiblePages: 5,
  showFirstLast: true,
  showPrevNext: true,
  showEllipsis: true,
  showInfo: true,
  size: 'default',    // 'sm', 'default', 'lg'
  rounded: false,
  align: 'center',    // 'start', 'center', 'end'
  icons: true,
  onChange: (pageData, prevPage) => {
    console.log('페이지 변경:', pageData);
  }
});

// 메서드
pagination.goToPage(5);           // 특정 페이지로 이동
pagination.setTotalItems(200);    // 총 아이템 수 변경
pagination.setItemsPerPage(20);   // 페이지당 아이템 수 변경
pagination.getPageData();         // 현재 페이지 데이터 가져오기
pagination.refresh();             // 새로고침
pagination.destroy();             // 정리
DataPaginator 클래스
// JSON 데이터와 함께 사용
const response = await fetch('data/paginationdata.json');
const data = await response.json();

const paginator = new Pagination.DataPaginator({
  container: '#container',
  data: data.users,
  itemsPerPage: 5,
  renderItem: (item, index) => {
    return `<div class="item">${item.name}</div>`;
  },
  onPageChange: (pageData, prevPage, items) => {
    console.log('현재 페이지 아이템:', items);
  },
  paginationOptions: {
    align: 'center',
    showInfo: true
  }
});

// 메서드
paginator.setData(newData);       // 데이터 변경
paginator.filter(item => item.status === '활성');  // 필터링
paginator.search('검색어', ['name', 'email']);     // 검색
paginator.sort((a, b) => a.price - b.price);       // 정렬
paginator.resetFilter();          // 필터 초기화
paginator.getCurrentPageData();   // 현재 페이지 데이터
paginator.goToPage(3);            // 페이지 이동
paginator.destroy();              // 정리

});

MultiSelect

const multiSelect = new Selectors.MultiSelect('#tags', {
  options: [
    { value: 'js', label: 'JavaScript' },
    { value: 'ts', label: 'TypeScript' }
  ],
  selected: ['js'],
  maxItems: 5,
  allowCreate: true,
  onChange: (values) => console.log(values)
});

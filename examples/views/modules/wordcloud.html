WordCloud (Canvas 기반)
const WordCloudModule = await IMCAT.use('wordcloud');

const wordcloud = new WordCloudModule.WordCloud('#container', {
  words: [
    { text: 'JavaScript', weight: 100 },
    { text: 'Python', weight: 80 },
    { text: 'React', weight: 70 },
  ],
  height: 400,
  minFontSize: 12,
  maxFontSize: 100,
  colors: ['#667eea', '#22c55e', '#f59e0b'],
  rotate: true,
  rotateAngles: [0, 90],      // 회전 각도
  maskIcon: 'favorite',        // 마스크: favorite, cloud, star, circle
  maskSize: 400,               // 마스크 크기
  shrinkToFit: true,           // 배치 실패 시 크기 줄여서 재시도
  onClick: (word, e) => {
    console.log('Clicked:', word.text, word.weight);
  }
});

// 이미지 다운로드
wordcloud.download('my-wordcloud.png');

// 메모리 관리
IMCAT.view.registerInstance(wordcloud);
TagCloud
const WordCloudModule = await IMCAT.use('wordcloud');

const tagcloud = new WordCloudModule.TagCloud('#container', {
  tags: [
    { text: 'Frontend', count: 45, color: '#667eea' },
    { text: 'Backend', count: 32 },
    { text: 'DevOps', count: 18 },
  ],
  variant: 'pill',          // default, pill, outlined, gradient
  size: 'md',               // sm, md, lg
  showCount: true,          // count 표시
  sizeByCount: true,        // count 기반 크기 조절
  removable: true,          // 삭제 버튼
  selectable: true,         // 선택 가능
  multiSelect: true,        // 다중 선택
  onClick: (tag, e) => {},
  onRemove: (tag, e) => {},
  onSelect: (selectedTags) => {}
});

IMCAT.view.registerInstance(tagcloud);
JSON 데이터 로드
// 1. fetch로 JSON 파일 로드
const response = await fetch('data/wordcloud-sample.json');
const words = await response.json();

const wordcloud = new WordCloud('#container', {
  words: words,
  height: 400
});

// 2. 파일 업로드로 로드
fileInput.onchange = (e) => {
  const file = e.target.files[0];
  const reader = new FileReader();
  
  reader.onload = (event) => {
    const data = JSON.parse(event.target.result);
    wordcloud.setWords(data);  // 단어 업데이트
  };
  
  reader.readAsText(file);
};

// JSON 형식 예시
// [
//   { "text": "JavaScript", "weight": 100 },
